 38:      * @param string クラスプレフィックス
 39:      */
 40:     public function addSearchPath( $dirname, $prefix = null )
 41:     {
 42:         // 配列添字キーを生成して、値をユニークにする。
 43:         $array_key = $dirname."|".$prefix;
 44:

 46:             'dirname'=>trim($dirname),
 47:             'prefix'=>trim($prefix)
 48:         );

 51:     /**
 52:      * 未定義クラスが呼ばれた時に呼ばれる
 53:      * オートロード関数
 54:      *
 55:      * @param string クラス名
 56:      * @param bool デバッグフラグ 初期値は偽
 57:      */
 58:     public function autoLoad( $name, $isDebug = false )
 59:     {
 60:         // クラスファイルの検索開始
 61:         foreach( $this->_search_path as $path )
 62:         {
 63:             // 検索候補かチェック
 64:             if( $this->_isPossibleSearchPath( $path, $name ) )
 65:             {
 66:                 // クラス名からパスを推定
 67:                 if( $found_file = $this->_guessFilePath( $path, $name ) )
 68:                 {
 69:                     // デバッグ中ならば
 70:                     // ファイルをロードしないでファイル名を返却
 71:                     if( $isDebug == false )
 72:                     {
 73:                         require_once $found_file;
 74:                         return true;
 75:                     }
 76:                     return $found_file;
 77:                 }
 78:             }
 79: 
 80:         }
 81:     }

 83:     /**
 84:      * クラスの完全修飾名からパスを推測する
 85:      *
 86:      * @param array path['prefix'],['dirname']
 87:      * @param string クラス名
 88:      */
 89:     private function _guessFilePath( $path, $name )
 90:     {
 91:         $path = $this->_applyPrefixPath( $path, $name );
 92: 
 93:         // クラス名からパスを生成
 94:         $path = str_replace(self::NAME_SPACE_SEPARATOR,'/',$path);
 95:         $path = str_replace(self::CLASS_NAME_SEPARATOR,'/',$path);
 96: 
 97:         foreach( array('','/class','/trait','/abstract','/interface') as $type )
 98:         {
 99:             if( file_exists( $file = sprintf('%s%s/%s.php',dirname($path), $type, basename($path) ) ) )
100:             {
101:                 return $file;
102:             }
103:         }
104:     }

106:     /** 
107:      * プレフィックスを登録されているパスに変換
108:      *
109:      * @param array path['prefix'],['dirname']
110:      * @param string クラス名
111:      */
112:     private function _applyPrefixPath( $path, $name )
113:     {
114:         $prefix = $path['prefix'];
115:         $dirname = $path['dirname'];
116: 
117:         if( empty($prefix) )
118:         {
119:             return sprintf('%s/%s',$dirname,$name);
120:         }
121:         // 変換用の正規表現を作成
122:         $regex = sprintf('/^%s/',preg_quote( $prefix ) );
123:         // プレフィックスをディレクトリ名に変更
124:         return preg_replace( $regex, $dirname, $name);
125:     }

127:     /**
128:      * 検索パスを絞り込む
129:      *
130:      * クラスが存在する可能性があれば真
131:      * プレフィックスが存在しなければ常に真
132:      *
133:      * @param array path 
134:      * @param string name
135:      */
136:     private function _isPossibleSearchPath( $path, $name )
137:     {
138:         $prefix = $path['prefix'];
139:         if( empty($prefix) )
140:         {
141:             return true;
142:         }
143: 
144:         if( 0 === substr_compare($name,$prefix,0,strlen($prefix)) )
145:         {
146:             return true;
147:         }
148:         return false;
149:     }
